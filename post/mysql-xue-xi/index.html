mysql学习

2020-01-08
  • 在安装好mysql的前提下（设置好开机自启动）
  • 校验mysql状态
 service mysqld status
  • 用户登录
mysql -u(用户名) -p(密码)
  • 查看数据库
 show databases;
  • 使用（切换）数据库
 use (数据库名称)
  • 查看表
 show tables
  • 查看表结构
desc 表名
  • 约束条件
create table 表名称(
    字段名称 数据类型[(长度) 约束条件1 约束条件2 ...],
    ...
);

1. unsigned - 无符号
2. zerofill - 0填充
3. not mull - 非空
4. default - 默认值
5. unique - 唯一
6. auto_increment - 自增
7. primary key - 主键
8. foreign key - 外键

  • 删除表 drop 表名；
  • 修改表结构
  #添加列       
  alter  table 表名 add 字段 类型(长度) 约束;
  #修改列        
  alter table 表名 modify 字段 类型(长度)约束；

9、删除表内容 delete from students;
delete from students where id=1;
